## Overview

The Chat components from `@px-ui/ai` provide building blocks for creating AI-powered conversational interfaces like chatbots and assistants.

## Import

```tsx
import {
  ChatContainer,
  ChatInput,
  MessageList,
  MessageBubble,
  TypingIndicator,
} from "@px-ui/ai";
```

## Components

### ChatContainer

A wrapper component that provides the layout structure for a chat interface.

```tsx
<ChatContainer className="h-[400px]">
  {/* Chat content */}
</ChatContainer>
```

### MessageList

Renders a scrollable list of messages with auto-scroll to the latest message.

```tsx
<MessageList
  messages={messages}
  autoScroll={true}
  renderMessage={({ message }) => (
    <MessageBubble
      role={message.role}
      content={message.content}
    />
  )}
/>
```

### MessageBubble

Displays an individual message with role-based styling (user, assistant, system).

```tsx
<MessageBubble role="user" content="Hello!" />
<MessageBubble role="assistant" content="Hi there! How can I help?" />
<MessageBubble role="system" content="System message" />
```

### ChatInput

A text input with send button for composing messages. Supports Enter to send and Shift+Enter for newlines.

```tsx
<ChatInput
  onSubmit={(value) => console.log(value)}
  isLoading={false}
  placeholder="Type a message..."
/>
```

### TypingIndicator

An animated indicator showing when the AI is "thinking".

```tsx
<TypingIndicator visible={isLoading} />
```

## Full Example

```tsx
import { useState } from "react";
import {
  ChatContainer,
  ChatInput,
  MessageList,
  TypingIndicator,
  type Message,
} from "@px-ui/ai";

function Chat() {
  const [messages, setMessages] = useState<Message[]>([]);
  const [isLoading, setIsLoading] = useState(false);

  const handleSubmit = async (value: string) => {
    // Add user message
    setMessages((prev) => [
      ...prev,
      { id: Date.now().toString(), role: "user", content: value },
    ]);

    // Call your AI backend here
    setIsLoading(true);
    const response = await fetchAIResponse(value);
    setMessages((prev) => [
      ...prev,
      { id: Date.now().toString(), role: "assistant", content: response },
    ]);
    setIsLoading(false);
  };

  return (
    <ChatContainer className="h-[500px]">
      <MessageList messages={messages} />
      {isLoading && <TypingIndicator />}
      <ChatInput onSubmit={handleSubmit} isLoading={isLoading} />
    </ChatContainer>
  );
}
```

## API Reference

### Message Type

```tsx
interface Message {
  id: string;
  role: "user" | "assistant" | "system";
  content: string;
  createdAt?: Date;
  metadata?: Record<string, unknown>;
}
```

### ChatContainer Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `className` | `string` | - | Additional CSS classes |
| `children` | `ReactNode` | - | Chat content |

### MessageList Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `messages` | `Message[]` | - | Array of messages to display |
| `autoScroll` | `boolean` | `true` | Auto-scroll to latest message |
| `renderMessage` | `(props: { message, index }) => ReactNode` | - | Custom message renderer |

### MessageBubble Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `role` | `"user" \| "assistant" \| "system"` | - | Message sender role |
| `content` | `ReactNode` | - | Message content |
| `renderContent` | `(content) => ReactNode` | - | Custom content renderer |

### ChatInput Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `onSubmit` | `(value: string) => void` | - | Called when message is sent |
| `isLoading` | `boolean` | `false` | Disables input while loading |
| `placeholder` | `string` | `"Type a message..."` | Input placeholder |
| `disabled` | `boolean` | `false` | Disable the input |

### TypingIndicator Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `visible` | `boolean` | `true` | Show/hide the indicator |
